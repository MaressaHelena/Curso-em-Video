```py
# >>>>>>>>>>>>>>>>>>>>>>> FUNÇÃO PRINT()


# O print() é responsável por mostrar coisas na tela, por exemplo:
print('Olá, Mundo!') # isso vai imprimir a mensagem Olá, Mundo!

# O print também pode nos ajudar para imprimir um cálculo, por exemplo:
print(7+4) # nesse caso não se usam aspas pois eu quero que me retorne um cálculo e não uma mensagem.

# Agora, vejamos um tipo de formatação da função print(), ela é utilizada para aproximar os elementos contidos no print.
print('7'+'4') # nessa situação, ao inves de imprimir 11 (que é a soma de 4+7), a função irá imprimir '74' pois o + ali significa APROXIMAÇÃO.
                # em alguns casos usa-se a vírgula para ''juntar'' os elementos.

#############################################################################################################################
# >>>>>>>>>>>>>>>>>>>>>>> VARIÁVEIS

# Tendo uma variável, devemos atribuir a ela ''objetos'', pois ela vai receber valores. Exemplo:
nome = 'Maressa Helena' # o igual (=), nesse caso, significa RECEBE! Lê-se: nome recebe Maressa Helena
idade = 20
peso = 50
# eu posso printar essas variaveis:
print(nome, idade, peso)

##############################################################################################################################
# >>>>>>>>>>>>>>>>>>>>>>> Função input()

# A função input() é para atribuir uma interação.

nome = input('Qual é seu nome?') # Nesse exemplo, eu vou perguntar o nome, a idade e o peso da pessoa e a mesma deverá inserir tais informações.
idade = input('Quantos anos você tem?')
peso = input('Qual é o seu peso?')
print(nome, idade, peso) #depois de realizada a interação e as variaveis guardarem as informações perguntadas, o programa irá imprimir todas elas, tal como está em print()
```
